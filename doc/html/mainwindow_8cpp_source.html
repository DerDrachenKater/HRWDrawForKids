205 void MainWindow::on_actionOpen_triggered()

208  // check if cancel was selected
209  if(fileName.isNull()) return;
210 
211  // open file
212  QFile file(fileName);
213  if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
214  qDebug() << "Unable to open file " << fileName << " for reading!"; // better implement a MessageBox for the user
215  return;
216  }
217 
218  // read from file
219  QJsonParseError jsonError;
220  QJsonDocument jsonDoc = QJsonDocument::fromJson(file.readAll(),&jsonError);
221  if(jsonError.error != QJsonParseError::NoError) {
222  qDebug() << "JSON Error: " << jsonError.errorString();
223  return;
224  }
225 
226  QJsonObject jsonObj = jsonDoc.object();
227 
228  // check app string
229  if(jsonObj["app"] != "HRW Vector Drawing") {
230  qDebug() << "This file is not from the HRW Vector Drawing app!"; // better implement a MessageBox for the user
231  return;
232  }
233 
234  // delete old objects
235  if(m_goMap) {
236  foreach(GraphicsObject *go, *m_goMap) {
237  delete go;
238  }
239  delete m_goMap;
240  }
241 
242  // read width and height but currently ignore the value as they are currently fixed
243  int width = jsonObj["width"].toInt();
244  int height = jsonObj["height"].toInt();
245 
246  // read graphics objects
247  QJsonArray jsonGraphicsObjects = jsonObj["graphicsObjects"].toArray();
248 
249  m_goMap = new GraphicsObjectsMap(jsonGraphicsObjects);
250  if(!m_goMap) {
251  qDebug() << "cannot allocate memory for goMap";
252  return;
253  }
254 
255  // add new objects to scene
256  foreach(GraphicsObject *go, *m_goMap) {
257  m_scene->addItem(go->graphicsItem());
258  }
259 
260  // close file
261  file.close();
